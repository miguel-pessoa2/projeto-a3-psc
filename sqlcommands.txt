create database projeto;
create table usuarios(
user_id int primary key auto_increment,
acesso enum("admin", "usuario") not null,
nome varchar(50) unique not null,
senha varchar(50) not null,
interesse1 enum("iaresponsavel", "ciberseguranca", "privacidade&etica"),
interesse2 enum("iaresponsavel", "ciberseguranca", "privacidade&etica"),
ativo boolean not null
);

create table recurso(
recurso_id int primary key auto_increment,
titulo varchar(100) not null,
autor varchar(100) not null,
descricao varchar(200),
categoria enum("iaresponsavel", "ciberseguranca", "privacidade&etica"),
comentario varchar(200),
link_recurso varchar(200) not null,
data_postagem timestamp default current_timestamp,
fk_user_id int,
foreign key (fk_user_id) references usuarios(user_id) on delete cascade
);
